<template>
  <div class="egova-mf-title-demo" style="min-height:30px;width:100%;padding:10px;">
      <a-config-provider :locale="locale">
         <mf-form :form-item-list="list" :rules="rules" layout="inline" :label-col="labelCol" :column="1" label-width="150px"></mf-form>
      </a-config-provider>
  </div>
</template>

<script>
import zh_CN from 'ant-design-vue/lib/locale-provider/zh_CN';
export default {
    data () {
        return {
            locale: zh_CN,
            labelCol:  { span: 6 },
            treeData: [
                { id: 1, pId: 0, value: '1', title: 'Expand to load' },
                { id: 2, pId: 0, value: '2', title: 'Expand to load' },
                { id: 3, pId: 0, value: '3', title: 'Tree Node', isLeaf: true },
            ],
            list: [
                { label: "工程名称", field: "projectName", type: "input" },
                { label: "工程地址", field: "address", type: "location", mapLocate: this.mapLocate },
                {
                    label: "工程类型",
                    field: "projectTypeId",
                    type: "select",
                    options: [{ id: 1, name: "工程类型" }],
                    select: this.typeSelect
                },
                { label: "填报日期", field: "reportDate", type: "rangePicker", options: [] },
                {
                    label: "所在地区",
                    field: "regionId",
                    type: "select",
                    options: [],
                    optionId: "regionId",
                    optionName: "regionName"
                },
                { label: "工程详情", field: "projectDetail", type: "textarea", maxLength: 20 },
                { label: "部门", field: "unit", type: "treeAsync", treeData: this.treeData, onLoadData: this.onLoadData },
            ],
            rules: {
                projectName: [{ required: true, trigger: "blur", message: "请输入工程名称" }],
                projectTypeId: [{ required: true, trigger: "blur", message: "请选择工程类型" }]
            },
        }
    },
    methods: {
        typeSelect(val) {
            if (val === 1) {
                this.list[3].options = [
                    { regionId: 1, regionName: "区域" }
                ];
            } else {
                this.list[3].options = [];
            }
        },
        onLoadData(file) {
        },
        mapLocate() {
        },
    }
}
</script>

# props

属性 说明 类型 传值 默认值
data 表单默认值 Object 非必传 -
uploadConfig 附件上传配置 Object 非必传 -
showUpload 显示附件上传 Boolean 非必传 true
column 每行列表数量 Number 非必传 3
formItemList 表单列表 Array 非必传 -
rules 表单校验规则 Object 非必传 -

# uploadConfig

属性 说明 类型 传值 默认值
label form-label显示名称 String 非必传 附件
defaultFiles 默认显示的文件，参考upload组件 Array 非必传 -
acceptFile 默认可接收的文件类型，参考upload组件 String 非必传 true

# column

属性 说明 类型 传值 默认值
label 字段显示名称 String 必传 -
field 字段属性 String 必传 -
type 表单类型 String，可选值有 select、textarea、yearPicker、datePicker、monthPicker、rangePicker、treeNode、treeCheck、treeAsync、location、phone、positiveNum、positiveInt、number 非必传 input
options 下拉选项，type为select有效 Array type为select必传 -
mode 下拉列表模式 default、multiple、tags、combobox 非必传 default
select type为select的选择值发生变化触发事件 Function 非必传 function(val, options)
optionId 指定options的key和value String 非必传 id
optionName 指定options的显示名称 String 非必传 name
treeData type为treeNode、treeCheck、treeAsync的选择树 String 非必传 -
onLoadData type为treeAsync的异步加载方法，具体查看ant-design-vue的异步选择树 Function 必传 -
mapLocate 点击定位触发 Function 非必传 -
years 年份选择列表，如[2023,2024,2025] Array 非必传 -
fieldRelateName 下拉显示名称数据 String 非必传 -
maxLength 输入框最大输入字符数 Number 非必传 input默认20，textarea默认150
column 当前formItem所占行数 String 非必传 1表示独占一行， 2:50，3:33.3%， 4: 25%
replaceFields 参考ant-design树的配置 Object 非必传 { title: 'text', key: 'id', value: 'id' }
