# 安装&使用

提示

Mf-ui 是基于 vue2 + ant-design-vue 二次封装的基础组件

# Install

npm install eg-mf-ui -S
&
yarn add eg-mf-ui -S

# Usage

// main.js
import ant from "ant-design-vue";
// 优先导入ant-design-vue
Vue.use(ant);

// 导入图片放大插件
import Viewer from "v-viewer";
Vue.use(Viewer);

import "eg-mf-ui/lib/mf-ui.css"
import Mfui from "eg-mf-ui"
Vue.use(Mfui)
